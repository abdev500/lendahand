1. Get the application URL by running these commands:
{{- if .Values.frontend.ingress.enabled }}
{{- range .Values.frontend.ingress.hosts }}
  http{{ if $.Values.frontend.ingress.tls }}s{{ end }}://{{ .host }}{{ .paths | first }}
{{- end }}
{{- else if contains "NodePort" .Values.frontend.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "lendahand.fullname" . }}-frontend)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.frontend.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           Watch the status with: 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ include "lendahand.fullname" . }}-frontend'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "lendahand.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.frontend.service.port }}
{{- else if contains "ClusterIP" .Values.frontend.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "lendahand.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=frontend" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

3. Access the admin panel:
   Backend admin: http://[backend-url]/admin
   Moderation dashboard: http://[backend-url]/moderation/

4. MinIO Storage (S3-compatible):
{{- if .Values.storage.minio.enabled }}
   MinIO is enabled for media file storage.
   MinIO is managed by the infra chart.
   Service: infra-minio
   API Port: 9000
   Console Port: 9001

   To access MinIO console:
   kubectl port-forward -n {{ .Release.Namespace }} svc/infra-minio 9001:9001
   Then visit: http://localhost:9001

   Note: MinIO configuration is in devops/infra/values.yaml
   Bucket name: lendahand-media (create via console or it will be auto-created on first upload)
{{- else }}
   MinIO storage is disabled. Media files are stored locally.
{{- end }}
